🎯 回答策略：从基础到高级 ​

回答框架（STAR 法则） ​

S - Situation: 什么项目，什么性能问题
T - Task: 你的优化目标  
A - Action: 具体采取的优化措施
R - Result: 优化后的效果数据

java
// 可以这样描述你的经验：
public class MemoryOptimizationExample {
    
    public void talkAboutMemoryOptimization() {
        /**
         * S: 在XX项目中，我们发现应用经常发生OOM崩溃
         * T: 需要降低内存使用，避免OOM
         * A: 我采取了以下措施：
         */
        
        // 1. 图片加载优化
        // - 使用Glide/Picasso替代直接Bitmap加载
        // - 配置合适的图片压缩和质量
        glide.load(imageUrl)
             .override(targetWidth, targetHeight)
             .format(DecodeFormat.PREFER_RGB_565)
             .diskCacheStrategy(DiskCacheStrategy.ALL);
        
        // 2. 内存泄漏检测
        // - 使用LeakCanary检测泄漏
        // - 修复Handler、静态引用等常见问题
        private static class SafeHandler extends Handler {
            private final WeakReference<Activity> mActivity;
            public SafeHandler(Activity activity) {
                mActivity = new WeakReference<>(activity);
            }
        }
        
        // 3. 大对象管理
        // - 使用对象池复用Bitmap
        // - 及时回收不再使用的资源
        bitmap.recycle();
        recyclerView.setRecycledViewPool(pool);
        
        /**
         * R: 优化后效果：
         * - OOM崩溃减少80%
         * - 平均内存使用降低30%
         * - 应用在低端设备更稳定
         */
    }
}

3. 启动速度优化 ​

java
public class StartupOptimizationExample {
    
    public void talkAboutStartupOptimization() {
        /**
         * S: 应用冷启动需要3-4秒，用户体验差
         * T: 将冷启动时间优化到2秒内
         * A: 启动优化措施：
         */
        
        // 1. 启动主题优化
        // - 使用启动窗口避免白屏
        <style name="AppTheme.Launcher">
            <item name="android:windowBackground">@drawable/launch_background</item>
        </style>
        
        // 2. 异步初始化
        // - 将非关键初始化移到后台
        appExecutors.diskIO().execute(() -> {
            // 数据库、文件等初始化
        });
        
        // 3. 延迟初始化
        // - 使用IdleHandler在空闲时初始化
        Looper.myQueue().addIdleHandler(() -> {
            initNonCriticalComponents();
            return false;
        });
        
        // 4. 启动过程监控
        // - 使用adb命令测量启动时间
        // adb shell am start -W -n package/.activity
        
        /**
         * R: 优化结果：
         * - 冷启动时间从3.5s降到1.8s
         * - 热启动时间从1.2s降到0.6s
         * - 用户留存率提升15%
         */
    }
}

4. 网络优化 ​

java
public class NetworkOptimizationExample {
    
    public void talkAboutNetworkOptimization() {
        /**
         * S: 应用在网络差时响应慢，图片加载卡顿
         * T: 提升弱网环境下的用户体验
         * A: 网络优化方案：
         */
        
        // 1. 请求合并与缓存
        // - 合并多个小请求为一个
        // - 合理设置缓存策略
        @Headers("Cache-Control: public, max-age=3600")
        @GET("data")
        Call<Data> getData();
        
        // 2. 图片加载优化
        // - 根据网络状况调整图片质量
        String imageUrl = NetworkUtils.isWifi(context) ? 
            highQualityUrl : lowQualityUrl;
        
        // 3. 重试机制
        // - 指数退避重试策略
        private void requestWithRetry(int retryCount) {
            if (retryCount < MAX_RETRY) {
                long delay = (long) Math.pow(2, retryCount) * 1000;
                handler.postDelayed(() -> makeRequest(retryCount + 1), delay);
            }
        }
        
        /**
         * R: 优化效果：
         * - 弱网环境下成功率从60%提升到85%
         * - 图片加载时间减少40%
         * - 用户流量消耗降低25%
         */
    }
}

5. 卡顿优化 ​

java
public class PerformanceOptimizationExample {
    
    public void talkAboutPerformanceOptimization() {
        /**
         * S: 应用在滚动列表和复杂动画时出现卡顿
         * T: 实现60fps的流畅体验
         * A: 性能优化措施：
         */
        
        // 1. 主线程监控
        // - 使用BlockCanary检测卡顿
        // - 监控每个消息的处理时间
        Looper.getMainLooper().setMessageLogging(printer -> {
            if (log.startsWith(">>>>> Dispatching")) {
                startTime = System.currentTimeMillis();
            } else if (log.startsWith("<<<<< Finished")) {
                long cost = System.currentTimeMillis() - startTime;
                if (cost > 16) logSlowOperation(cost);
            }
        });
        
        // 2. 渲染优化
        // - 使用硬件层加速动画
        view.setLayerType(View.LAYER_TYPE_HARDWARE, null);
        // 动画完成后切回
        view.setLayerType(View.LAYER_TYPE_NONE, null);
        
        // 3. 内存抖动优化
        // - 避免在循环中创建对象
        // - 使用对象池复用
        private final Pools.SimplePool<Bitmap> bitmapPool = 
            new Pools.SimplePool<>(10);
        
        /**
         * R: 优化成果：
         * - 列表滚动帧率从45fps提升到58fps
         * - 卡顿次数减少90%
         * - 动画流畅度显著改善
         */
    }
}
