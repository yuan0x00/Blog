1. 单例模式 (Singleton) - ⭐最常用 ​

java
// Android中的典型应用
public class SingletonExamples {
    
    // 方式1：双重校验锁
    public class NetworkManager {
        private static volatile NetworkManager instance;
        
        private NetworkManager() {
            // 私有构造函数
        }
        
        public static NetworkManager getInstance() {
            if (instance == null) {
                synchronized (NetworkManager.class) {
                    if (instance == null) {
                        instance = new NetworkManager();
                    }
                }
            }
            return instance;
        }
    }
    
    // 方式2：静态内部类（推荐）
    public class ImageLoader {
        private ImageLoader() {}
        
        private static class Holder {
            private static final ImageLoader INSTANCE = new ImageLoader();
        }
        
        public static ImageLoader getInstance() {
            return Holder.INSTANCE;
        }
    }
    
    // Android系统中的应用：
    // - LayoutInflater.from(context)
    // - AccessibilityManager.getInstance(context)
    // - 各种Manager类
}

2. 建造者模式 (Builder) - ⭐Android大量使用 ​

java
public class BuilderPatternExamples {
    
    // Android中的典型应用
    public class AndroidBuilderExamples {
        /**
         * 1. AlertDialog.Builder
         * 2. Notification.Builder  
         * 3. Retrofit.Builder
         * 4. Glide.with().load().into()
         */
    }
    
    // 自定义Builder示例
    public class User {
        private final String name;
        private final int age;
        private final String email;
        
        private User(Builder builder) {
            this.name = builder.name;
            this.age = builder.age;
            this.email = builder.email;
        }
        
        public static class Builder {
            private String name;
            private int age;
            private String email;
            
            public Builder setName(String name) {
                this.name = name;
                return this;
            }
            
            public Builder setAge(int age) {
                this.age = age;
                return this;
            }
            
            public Builder setEmail(String email) {
                this.email = email;
                return this;
            }
            
            public User build() {
                return new User(this);
            }
        }
    }
    
    // 使用方式
    User user = new User.Builder()
        .setName("张三")
        .setAge(25)
        .setEmail("zhangsan@example.com")
        .build();
}

3. 工厂模式 (Factory) ​

public class FactoryPatternExamples {
    
    // 简单工厂
    public class ViewFactory {
        public static View createView(Context context, String type) {
            switch (type) {
                case "button":
                    return new Button(context);
                case "textview":
                    return new TextView(context);
                case "imageview":
                    return new ImageView(context);
                default:
                    throw new IllegalArgumentException("Unknown view type");
            }
        }
    }
    
    // Android中的应用：

    // - BitmapFactory：根据资源创建Bitmap

        Bitmap bitmap = BitmapFactory.decodeResource(getResources(), R.drawable.image);
    }
}

4. 适配器模式 (Adapter) - ⭐RecyclerView核心 ​

java
public class AdapterPatternExamples {
    
    // RecyclerView.Adapter就是典型的适配器模式
    public class UserAdapter extends RecyclerView.Adapter<UserAdapter.ViewHolder> {
        private List<User> mUsers;
        
        public UserAdapter(List<User> users) {
            mUsers = users;
        }
        
        @NonNull
        @Override
        public ViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
            // 创建View
            View view = LayoutInflater.from(parent.getContext())
                .inflate(R.layout.item_user, parent, false);
            return new ViewHolder(view);
        }
        
        @Override
        public void onBindViewHolder(@NonNull ViewHolder holder, int position) {
            // 绑定数据到View
            User user = mUsers.get(position);
            holder.bind(user);
        }
        
        @Override
        public int getItemCount() {
            return mUsers.size();
        }
        
        class ViewHolder extends RecyclerView.ViewHolder {
            // ViewHolder模式也是设计模式的一种
            private TextView nameText;
            private ImageView avatarImage;
            
            public ViewHolder(@NonNull View itemView) {
                super(itemView);
                nameText = itemView.findViewById(R.id.tv_name);
                avatarImage = itemView.findViewById(R.id.iv_avatar);
            }
            
            public void bind(User user) {
                nameText.setText(user.getName());
                // 加载图片等
            }
        }
    }
    
    // 其他Android应用：
    // - ArrayAdapter, CursorAdapter
    // - ListView的Adapter
}

6. 外观模式 (Facade) ​

java
public class FacadePatternExamples {
    
    // Glide就是典型的外观模式
    public class ImageLoaderFacade {
        // 复杂的图片加载过程被封装成简单接口
        Glide.with(context)
            .load(url)
            .placeholder(R.drawable.placeholder)
            .error(R.drawable.error)
            .into(imageView);
    }
    
    // Retrofit也是外观模式
    public interface ApiService {
        @GET("users/{id}")
        Call<User> getUser(@Path("id") String userId);
    }
    
    // 使用简单接口隐藏复杂实现
    Retrofit retrofit = new Retrofit.Builder()
        .baseUrl("https://api.example.com/")
        .build();
    
    ApiService service = retrofit.create(ApiService.class);
    Call<User> call = service.getUser("123");
}

🔄 行为型模式 (Behavioral Patterns) ​

7. 观察者模式 (Observer) - ⭐LiveData核心 ​

java
public class ObserverPatternExamples {
    
    // LiveData/ViewModel的观察者模式
    public class UserViewModel extends ViewModel {
        private MutableLiveData<User> userLiveData = new MutableLiveData<>();
        
        public LiveData<User> getUser() {
            return userLiveData;
        }
        
        public void updateUser(User user) {
            userLiveData.setValue(user);
        }
    }
    
    // Activity中观察数据变化
    public class ProfileActivity extends AppCompatActivity {
        @Override
        protected void onCreate(Bundle savedInstanceState) {
            super.onCreate(savedInstanceState);
            
            UserViewModel viewModel = new ViewModelProvider(this).get(UserViewModel.class);
            
            // 观察数据变化，自动更新UI
            viewModel.getUser().observe(this, new Observer<User>() {
                @Override
                public void onChanged(User user) {
                    updateUI(user);
                }
            });
        }
    }
    
    // 其他Android观察者应用：
    // - BroadcastReceiver
    // - EventBus
    // - RxJava Observable
}

8. 策略模式 (Strategy) ​

java
public class StrategyPatternExamples {
    
    // 图片加载策略
    public interface ImageCacheStrategy {
        void loadImage(String url, ImageView imageView);
    }
    
    public class MemoryCacheStrategy implements ImageCacheStrategy {
        @Override
        public void loadImage(String url, ImageView imageView) {
            // 内存缓存加载逻辑
        }
    }
    
    public class DiskCacheStrategy implements ImageCacheStrategy {
        @Override
        public void loadImage(String url, ImageView imageView) {
            // 磁盘缓存加载逻辑
        }
    }
    
    public class NetworkLoadStrategy implements ImageCacheStrategy {
        @Override
        public void loadImage(String url, ImageView imageView) {
            // 网络加载逻辑
        }
    }
    
    // 使用策略
    public class ImageLoader {
        private ImageCacheStrategy strategy;
        
        public void setStrategy(ImageCacheStrategy strategy) {
            this.strategy = strategy;
        }
        
        public void load(String url, ImageView imageView) {
            strategy.loadImage(url, imageView);
        }
    }
}

9. 模板方法模式 (Template Method) ​

java
public class TemplateMethodExamples {
    
    // AsyncTask就是模板方法模式
    public class MyAsyncTask extends AsyncTask<String, Integer, String> {
        
        @Override
        protected void onPreExecute() {
            // 执行前准备（主线程）
            showProgressDialog();
        }
        
        @Override
        protected String doInBackground(String... params) {
            // 后台执行（工作线程）
            return performLongRunningTask(params[0]);
        }
        
        @Override
        protected void onProgressUpdate(Integer... values) {
            // 进度更新（主线程）
            updateProgress(values[0]);
        }
        
        @Override
        protected void onPostExecute(String result) {
            // 执行完成（主线程）
            hideProgressDialog();
            showResult(result);
        }
    }
    
    // Activity的生命周期也是模板方法
    public class BaseActivity extends AppCompatActivity {
        // onCreate, onStart, onResume等由系统调用模板
    }
}

10. MVC / MVP / MVVM - 架构模式 ​

java
public class ArchitecturePatterns {
    
    // MVVM with ViewModel + LiveData
    public class MVVMExample {
        /**
         * Model: Data层 (Repository, DataSource)
         * View: Activity/Fragment (只负责UI)
         * ViewModel: 业务逻辑，连接View和Model
         */
    }
    
    // MVP示例
    public class MVPExample {
        public interface LoginContract {
            interface View {
                void showLoading();
                void hideLoading();
                void onLoginSuccess();
                void onLoginFailed(String error);
            }
            
            interface Presenter {
                void login(String username, String password);
            }
        }
        
        public class LoginPresenter implements LoginContract.Presenter {
            private LoginContract.View view;
            
            public LoginPresenter(LoginContract.View view) {
                this.view = view;
            }
            
            @Override
            public void login(String username, String password) {
                view.showLoading();
                // 处理登录逻辑
            }
        }
    }
}

public class DependencyInjectionExamples {

            super.onCreate(savedInstanceState);
            // Dagger会自动注入retrofit实例
        }
    }
